# Create a table for mean purchase and CI
clinic_summary <- clinic %>%  
  mutate(treatment = as.factor(treatment)) %>% 
  group_by(treatment) %>%       
  summarise(clinic_number = length(clinicnumber),            
            clinic_avgnpurchase = round(mean(purchase),3),         
            clinic_error = round(sd(purchase)/sqrt(clinic_number),3),  
            lowerCI = round(clinic_avgnpurchase - 1.96*clinic_error,2),   
            upperCI = round(clinic_avgnpurchase + 1.96*clinic_error,2))

clinic_ATE <-matrix(NA,ncol = 3, nrow = 3)  # create a matrix to store the results
colnames(clinic_ATE) <- c("Treatment Effect units purchased", "Lower 95% CI", "Upper 95% CI" )
rownames(clinic_ATE) <- c("Mean Easier to use", "Fewer errors", "Control Mean:cost-focused")

clinic_effect <-c(clinic_summary$clinic_avgnpurchase[2]-clinic_summary$clinic_avgnpurchase[1],
                  clinic_summary$clinic_avgnpurchase[3]-clinic_summary$clinic_avgnpurchase[1],
                  clinic_summary$clinic_avgnpurchase[1]) 

clinic_error_rate <- c (sqrt(clinic_summary$clinic_error[1]^2+clinic_summary$clinic_error[2]^2),
                        sqrt(clinic_summary$clinic_error[1]^2+clinic_summary$clinic_error[3]^2), NA)
# calculate the s.e. of ATE
clinic_LCI <- clinic_effect-1.96*clinic_error_rate
clinic_UCI <- clinic_effect+1.96*clinic_error_rate

clinic_ATE[,1] <-round(clinic_effect,3)
clinic_ATE[,2] <-round(clinic_LCI,3)
clinic_ATE[,3] <-round(clinic_UCI,3)
